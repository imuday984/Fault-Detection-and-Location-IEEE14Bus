import pandas as pd
import numpy as np

from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler

from sklearn.metrics import (
    accuracy_score,
    confusion_matrix,
    classification_report
)

from xgboost import XGBClassifier

# =====================================
# LOAD DATASET
# =====================================

df = pd.read_csv("LG_2058_Dataset.csv")

print("Dataset Shape:", df.shape)

# =====================================
# FEATURES & LABELS
# =====================================

X = df.drop("FaultBus", axis=1)

y = df["FaultBus"]

# convert 1-14 → 0-13

y = y - 1

# =====================================
# TRAIN TEST SPLIT
# =====================================

X_train, X_test, y_train, y_test = train_test_split(
    X,
    y,
    test_size=0.25,
    random_state=42,
    stratify=y
)

# =====================================
# SCALING
# =====================================

scaler = StandardScaler()

X_train = scaler.fit_transform(X_train)

X_test = scaler.transform(X_test)

# =====================================
# XGBOOST MODEL
# =====================================

model = XGBClassifier(
    n_estimators=500,
    max_depth=8,
    learning_rate=0.03,
    subsample=0.8,
    colsample_bytree=0.8,
    objective='multi:softmax',
    num_class=14,
    random_state=42
)

# =====================================
# TRAIN
# =====================================

model.fit(X_train, y_train)

# =====================================
# PREDICT
# =====================================

y_pred = model.predict(X_test)

# =====================================
# RESULTS
# =====================================

acc = accuracy_score(y_test, y_pred)

print("\n========================")
print("TEST ACCURACY")
print("========================")
print(f"{acc*100:.2f}%")

print("\n========================")
print("CONFUSION MATRIX")
print("========================")
print(confusion_matrix(y_test, y_pred))

print("\n========================")
print("CLASSIFICATION REPORT")
print("========================")
print(classification_report(y_test, y_pred))

results = pd.DataFrame({
    "Actual": y_test,
    "Predicted": y_pred
})

print("\n========================")
print("ACTUAL VS PREDICTED")
print("========================")
print(results.head(50))

from sklearn.ensemble import ExtraTreesClassifier

# Re-split with 20% test size to give the model more training data
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.20, random_state=42, stratify=y
)

# Using the scaled features
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train)
X_test_scaled = scaler.transform(X_test)

# Initialize Extra Trees with optimized parameters
et_model = ExtraTreesClassifier(
    n_estimators=1000,
    max_depth=None,
    min_samples_split=2,
    random_state=42,
    n_jobs=-1
)

# Train
et_model.fit(X_train_scaled, y_train)

# Predict
y_pred_et = et_model.predict(X_test_scaled)

# Evaluate
accuracy_et = accuracy_score(y_test, y_pred_et)
print(f"Extra Trees Test Accuracy: {accuracy_et*100:.2f}%")

if accuracy_et < 0.96:
    print("Target not met yet. Refining model...")
else:
    print("Target reached!")
